function getShortestPathCmdGreedy(pos1, pos2) {
  var columnDistance = Math.abs(pos1.c - pos2.c);
  var rowDistance = Math.abs(pos1.r - pos2.r);

  if (rowDistance < columnDistance && rowDistance > 0) {
    if (pos1.r > pos2.r && Maze[pos1.r - 1][pos1.c] == "")
      return "U";
    else if (pos1.r < pos2.r && Maze[pos1.r + 1][pos1.c] == "")
      return "D";
  }
  else if (columnDistance > 0) {
    if (pos1.c > pos2.c && Maze[pos1.r][pos1.c - 1] == "")
      return "L";
    else if (pos1.c < pos2.c && Maze[pos1.r][pos1.c + 1] == "")
      return "R";
  }
  return "";
}

function getNewCommand() {
  for (var i = 0; i < Tanks.length; i += 1) {
    var t = Tanks[i];
    if (t.color != "white") continue;

    // white tank is on the right
    if (MyTank.r == t.r && t.c > MyTank.c) {
      if ( !isPathBlocked(MyTank, t) ) {
        if (MyTank.dir == "R") return "S";
        else return "R";
      }
    }

    // white tank is on the left
    if (MyTank.r == t.r && t.c < MyTank.c) {
      if ( !isPathBlocked(t, MyTank) ) {
        if (MyTank.dir == "L") return "S";
        else return "L";
      }
    }

    // white tank is below
    if (MyTank.c == t.c && t.r > MyTank.r) {
      if ( !isPathBlockedVertical(MyTank, t) ) {
        if (MyTank.dir == "D") return "S";
        else return "D";
      }
    }
  
    // white tank is above
    if (MyTank.c == t.c && t.r < MyTank.r) {
      if ( !isPathBlockedVertical(t, MyTank) ) {
        if (MyTank.dir == "U") return "S";
        else return "U";
      }
    }

    // move to white tank's row or column
    var columnDistance = Math.abs(MyTank.c - t.c);
    var rowDistance = Math.abs(MyTank.r - t.r);

    if (rowDistance < columnDistance) {
      if (MyTank.r > t.r) {
        // new logic: would go up only if it is not a blocking tile
        if (Maze[MyTank.r-1][MyTank.c] == "") return "U";
      } else {
        // check if the tile below is not blocking
        if ( Maze[MyTank.r+1][MyTank.c] == "" ) return "D";
      }
    } else { 
      if (MyTank.c > t.c) {
        // check if the tile on the left is not blocking
        if ( Maze[MyTank.r][MyTank.c-1] == "" ) return "L";
      } else {
        // check if the tile on the right is not blocking
        if ( Maze[MyTank.r][MyTank.c+1] == "" ) return "R";
      }
    }
  }
  return getRandomCommand();
}
