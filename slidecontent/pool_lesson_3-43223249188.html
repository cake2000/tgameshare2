// define the function
function calculateAverage(n1, n2, n3) {
  var total = n1 + n2 + n3;
  var average = total / 3;
  return average;
}

// execute the function
var avg = calculateAverage(5, 6, 7);
// initialize parameters
var n1 = 5;
var n2 = 6;
var n3 = 7;

// function body
var total = n1 + n2 + n3;
var average = total / 3;
var output = average;

// saving output 
var avg = output;
