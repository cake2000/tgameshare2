async function getCommand(pocketID, ballID) {
  var aim = extrapolatePoints(Pockets[pocketID], Balls[ballID], BallDiameter);
  var cmd = {
    aimx: aim.x, aimy: aim.y, strength: 35, targetPocketID: pocketID, targetBallID: ballID
  };  
  cmd.probability = await calculateProbability(cmd);
  print("pocket " + pocketID + " ball " + ballID + " probability " + cmd.probability);
  return cmd;
}
