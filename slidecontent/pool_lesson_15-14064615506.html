async function getKickShot() {
   const targetBallPos = Balls[4];
   // calculate the contact position for kicking ball 4 into pocket 3
   const contactPos = extrapolatePoints(Pockets[3], Balls[4], BallDiameter);
   const mirrorPos = {x: contactPos.x, y: 2 * TOP_Y - contactPos.y};
   const cmd = { aimx: mirrorPos.x, aimy: mirrorPos.y, strength: 40 };

   // calculate closest trajectory point
   const closestPoint = await calculateCTP(cmd, 0, contactPos);
   const errorInX = Math.abs(closestPoint.x - contactPos.x);
   if (closestPoint.x > contactPos.x) {
    cmd.aimx = cmd.aimx - errorInX;
