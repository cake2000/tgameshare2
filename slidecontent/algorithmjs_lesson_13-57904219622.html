function hasCycle(N, graph) {
  const visited = new Array(N).fill(0);
  const path = new Array(N).fill(0);
  for (let i = 0; i < N; i += 1) {
    if (hasCycleUtil(graph, visited, path, 0)) return true;
  }
  return false;
}
