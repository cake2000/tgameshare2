function bubbleSort(arrayOfNums) {
  for (let i = 0; i < arrayOfNums.length - 1; i += 1) {
    for (let j = 0; j < arrayOfNums.length - i - 1; j += 1) {
      if (arrayOfNums[j] > arrayOfNums[j + 1]) {
        swapElements(arrayOfNums, j, j + 1);
      }
    }
  }
}

function selectionSort(arrayOfNums) {
  for (let i = 0; i < arrayOfNums.length - 1; i += 1) {
    const idx = getSmallest(arrayOfNums, i);
    if (i != idx) swapElements(arrayOfNums, i, idx);
  }
}
